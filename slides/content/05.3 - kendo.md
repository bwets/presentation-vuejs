## Kendo

```html
<template>
	<kendo-grid :id="'grid'" :height="'100%'" :data-source="items" :sortable="true">
		<kendo-grid-column :command="commands"></kendo-grid-column>
		<kendo-grid-column field="status" title="Status"></kendo-grid-column>
		<kendo-grid-column field="name" title="Name"></kendo-grid-column>
	</kendo-grid>
<template>
<script lang="ts"> 
	@Component()
	export default class ItemsList extends Vue {
		items = [ 
			{status:"Done", name="Item 1"},
			{status:"Pending", name="Item 2"}
		];

		get commands() {
			return [
				{ name: "details", text: "", click: this.view_click, iconClass: "k-icon k-i-copy" },
				{ name: "delete", text: "", click: this.delete_click, iconClass: "k-icon k-i-close" }
			];
		}
	}
</script>
```
